\begin{algorithm}[htb]
	%\footnotesize
	\label{alg:bfs}
	\caption{Breadth-first search.}
	%\KwData{...}
	%\KwResult{...}
	\Fn{BFS}{
		$\grbfrontier(\grbsource) = \grbtrue$ \;
		\For{$\grblevel = 1$ \KwTo $\grbs{n}-1$}{
			$\grbseen \grbmask{\grbfrontier} = \grblevel$ \;
			$\grbfrontier \grbmaskreplace{\grbneg \grbseen} = \grbfrontier \grbanypair \grbmA$ \;
		}
	}
\end{algorithm}

\begin{algorithm}[htb]
	\label{alg:bfs}
	\caption{Breadth-first search (push/pull).}
	%\KwData{...}
	%\KwResult{...}
	\Fn{BFS}{
		$\grbfrontier(\grbs{source}) = \grbtrue$ \;
		\For{$\grblevel = 1$ \KwTo $\grbs{n}-1$}{
			$\grbseen \grbmask{\grbfrontier} = \grblevel$ \;
			
			\If{$\mathit{TraversalShouldPush}(\grbmA, \grbfrontier)$}{ %\Comment*{Decide to push/pull}
				$\grbfrontier \grbmaskreplace{\grbneg \grbseen} = \grbfrontier \grbanypair \grbmA$
			}
			\Else{
				$\grbfrontier \grbmaskreplace{\grbneg \grbseen} = \grbmA \grbanypair \grbfrontier$
			}
		}
	}
\end{algorithm}

\begin{algorithm}[htb]
	\label{alg:msbfs}
	\caption{Multi-source breadth-first search.}
	\DontPrintSemicolon
	\KwData{...}
	\KwResult{...}
	\Fn{MSBFS}{
		$\grbFrontier = \mathsf{diag}(\grba{sources}, \grbs{n})$ \;
		\For{$\grblevel = 1$ \textbf{to} $n-1$}{
			$\grbSeen\grbmask{\grbFrontier} = \grblevel$ \;
			$\grbFrontier\grbmaskreplace{\grbneg \grbSeen} = \grbFrontier \grbanypair \grbm{A} $ \;
		}
	}
\end{algorithm}
