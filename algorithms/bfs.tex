\begin{algorithm}[htb]
	\label{alg:bfs-levels}
	\caption{BFS / Levels variant.}
	\DontPrintSemicolon
	\KwIn{$\grbm{A}, \grbs{n}, \grbs{startVertex}$}
	\Fn{BFS}{
		$\grbv{f}(\grbs{startVertex}) = \grbT$ \;

		\For{$\grblevel = 1$ \KwTo $\grbs{n}-1$}{
			$\grbv{s} \grbmask{\grbv{f}} = \grblevel$ \;

			$\grbv{f} \grbmaskreplace{\grbneg \grbv{s}} = \grbv{f} %\grbanypair
			\grbmA$ \;
		}
	}
\end{algorithm}

\begin{algorithm}[htb]
	\label{alg:bfs-parents}
	\caption{BFS / Parents variant.}
	\DontPrintSemicolon
	\KwIn{$\grbm{A}, \grbs{n}, \grbs{startVertex}$}
	\Fn{ParentsBFS}{
		$\grbv{f}(\grbs{startVertex}) = 0$ \;

		\For{$\grblevel = 1$ \KwTo $\grbs{n}-1$}{
			$\grbv{s} \grbmask{\grbv{f}} = \grbv{f}$ \;

			$\grbv{f} \grbmaskreplace{\grbneg \grbv{s}} = \grbv{f} \grbanyfirstjone \ \grbmA$ \;
		}
	}
\end{algorithm}

\begin{algorithm}[htb]
	\label{alg:bfs-push-pull}
	\caption{Direction Optimizing (push/pull) BFS.}
	\DontPrintSemicolon
	\KwIn{$\grbm{A}, \grbm{A}\grbt, \grbs{n}, \grbs{startVertex}$}
	\Fn{DirectionOptimizingBFS}{
		$\grbv{f}(\grbs{startVertex}) = \grbT$ \;

		\For{$\grblevel = 1$ \KwTo $\grbs{n}-1$}{
			$\grbv{s} \grbmask{\grbv{f}} = \grblevel$ \;

			\If{$\mathit{Push}(\grbmA, \grbv{f})$}{ %\Comment*{Decide to push/pull}
				$\grbv{f} \grbmaskreplace{\grbneg \grbv{s}} = \grbv{f} \grbmA$
			}
			\Else{
				$\grbv{f} \grbmaskreplace{\grbneg \grbv{s}} = \grbmA\grbt \grbv{f}$
			}
		}
	}
\end{algorithm}
